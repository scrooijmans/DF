  import { z } from &quot;zod/v4&quot;;
  const formSchema = z.object({
    pin: z.string().min(6, {
      message: &quot;Your one-time password must be at least 6 characters.&quot;
    })
  });
&lt;/script&gt;
 
&lt;script lang=&quot;ts&quot;&gt;
  import { defaults, superForm } from &quot;sveltekit-superforms&quot;;
  import { zod4 } from &quot;sveltekit-superforms/adapters&quot;;
  import { toast } from &quot;svelte-sonner&quot;;
  import * as InputOTP from &quot;$lib/components/ui/input-otp/index.js&quot;;
  import * as Form from &quot;$lib/components/ui/form/index.js&quot;;
 
  const form = superForm(defaults(zod4(formSchema)), {
    validators: zod4(formSchema),
    SPA: true,
    onUpdate: ({ form: f }) =&gt; {
      if (f.valid) {
        toast.success(`You submitted ${JSON.stringify(f.data, null, 2)}`);
      } else {
        toast.error(&quot;Please fix the errors in the form.&quot;);
      }
    }
  });
 
  const { form: formData, enhance } = form;
&lt;/script&gt;
 
&lt;form method=&quot;POST&quot; class=&quot;w-2/3 space-y-6&quot; use:enhance&gt;
  &lt;Form.Field {form} name=&quot;pin&quot;&gt;
    &lt;Form.Control&gt;
      {#snippet children({ props })}
        &lt;InputOTP.Root maxlength={6} {...props} bind:value={$formData.pin}&gt;
          {#snippet children({ cells })}
            &lt;InputOTP.Group&gt;
              {#each cells as cell (cell)}
                &lt;InputOTP.Slot {cell} /&gt;
              {/each}
            &lt;/InputOTP.Group&gt;
          {/snippet}
        &lt;/InputOTP.Root&gt;
      {/snippet}
    &lt;/Form.Control&gt;
    &lt;Form.Description
      &gt;Please enter the one-time password sent to your phone.&lt;/Form.Description
    &gt;
    &lt;Form.FieldErrors /&gt;
  &lt;/Form.Field&gt;
  &lt;Form.Button&gt;Submit&lt;/Form.Button&gt;
&lt;/form&gt;</code></pre>
