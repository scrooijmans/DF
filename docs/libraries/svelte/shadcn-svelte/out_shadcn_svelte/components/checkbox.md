  import { z } from &quot;zod/v4&quot;;
 
  const items = [
    {
      id: &quot;recents&quot;,
      label: &quot;Recents&quot;
    },
    {
      id: &quot;home&quot;,
      label: &quot;Home&quot;
    },
    {
      id: &quot;applications&quot;,
      label: &quot;Applications&quot;
    },
    {
      id: &quot;desktop&quot;,
      label: &quot;Desktop&quot;
    },
    {
      id: &quot;downloads&quot;,
      label: &quot;Downloads&quot;
    },
    {
      id: &quot;documents&quot;,
      label: &quot;Documents&quot;
    }
  ] as const;
 
  const formSchema = z.object({
    items: z.array(z.string()).refine((value) =&gt; value.some((item) =&gt; item), {
      message: &quot;You have to select at least one item.&quot;
    })
  });
&lt;/script&gt;
 
&lt;script lang=&quot;ts&quot;&gt;
  import { defaults, superForm } from &quot;sveltekit-superforms&quot;;
  import { zod4 } from &quot;sveltekit-superforms/adapters&quot;;
  import { toast } from &quot;svelte-sonner&quot;;
  import * as Form from &quot;$lib/components/ui/form/index.js&quot;;
  import { Checkbox } from &quot;$lib/components/ui/checkbox/index.js&quot;;
 
  const form = superForm(defaults(zod4(formSchema)), {
    SPA: true,
    validators: zod4(formSchema),
    onUpdate: ({ form: f }) =&gt; {
      if (f.valid) {
        toast.success(`You submitted ${JSON.stringify(f.data, null, 2)}`);
      } else {
        toast.error(&quot;Please fix the errors in the form.&quot;);
      }
    }
  });
 
  const { form: formData, enhance } = form;
 
  function addItem(id: string) {
    $formData.items = [...$formData.items, id];
  }
 
  function removeItem(id: string) {
    $formData.items = $formData.items.filter((i) =&gt; i !== id);
  }
&lt;/script&gt;
 
&lt;form method=&quot;POST&quot; class=&quot;space-y-8&quot; use:enhance&gt;
  &lt;Form.Fieldset {form} name=&quot;items&quot; class=&quot;space-y-0&quot;&gt;
    &lt;div class=&quot;mb-4&quot;&gt;
      &lt;Form.Legend class=&quot;text-base&quot;&gt;Sidebar&lt;/Form.Legend&gt;
      &lt;Form.Description&gt;
        Select the items you want to display in the sidebar.
      &lt;/Form.Description&gt;
    &lt;/div&gt;
    &lt;div class=&quot;space-y-2&quot;&gt;
      {#each items as item (item.id)}
        {@const checked = $formData.items.includes(item.id)}
        &lt;div class=&quot;flex flex-row items-start space-x-3&quot;&gt;
          &lt;Form.Control&gt;
            {#snippet children({ props })}
              &lt;Checkbox
                {...props}
                {checked}
                value={item.id}
                onCheckedChange={(v) =&gt; {
                  if (v) {
                    addItem(item.id);
                  } else {
                    removeItem(item.id);
                  }
                }}
              /&gt;
              &lt;Form.Label class=&quot;font-normal&quot;&gt;
                {item.label}
              &lt;/Form.Label&gt;
            {/snippet}
          &lt;/Form.Control&gt;
        &lt;/div&gt;
      {/each}
      &lt;Form.FieldErrors /&gt;
    &lt;/div&gt;
  &lt;/Form.Fieldset&gt;
  &lt;Form.Button&gt;Update display&lt;/Form.Button&gt;
