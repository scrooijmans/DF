Adding dark mode to your Astro site.

Just like in regular Svelte, we use the `class` strategy from Tailwind CSS to support dark mode toggling. See the <a href="https://tailwindcss.com/docs/dark-mode#toggling-dark-mode-manually" class="font-medium underline underline-offset-4" rel="noopener noreferrer" target="_blank">Tailwind CSS documentation</a> for more information.

How you add the `dark` class to the `html` element is up to you. In this guide, we'll take a look at enabling dark mode toggling with <a href="https://github.com/svecosystem/mode-watcher" class="font-medium underline underline-offset-4" rel="noopener noreferrer" target="_blank">mode-watcher</a>.

## Usage

### Create an inline theme script

This script will, in part, keep and track the dark mode value in `localStorage` and prevent <a href="https://en.wikipedia.org/wiki/Flash_of_unstyled_content" class="font-medium underline underline-offset-4" rel="noopener noreferrer" target="_blank">FUOC</a>.

<figure data-metadata="" data-rehype-pretty-code-figure="">
<pre class="no-scrollbar" data-language="astro" data-metadata="" data-theme="github-dark github-light-default" tabindex="0"><code>---
import &quot;../styles/global.css&quot;;
---
 
&lt;script is:inline&gt;
  const isBrowser = typeof localStorage !== &#39;undefined&#39;;
  const getThemePreference = () =&gt; {
    if (isBrowser &amp;&amp; localStorage.getItem(&#39;theme&#39;)) {
      return localStorage.getItem(&#39;theme&#39;);
    }
    return window.matchMedia(&#39;(prefers-color-scheme: dark)&#39;).matches
      ? &#39;dark&#39; : &#39;light&#39;;
  };
  const isDark = getThemePreference() === &#39;dark&#39;;
  document.documentElement.classList[isDark ? &#39;add&#39; : &#39;remove&#39;](&#39;dark&#39;);
 
  if (isBrowser) {
    const observer = new MutationObserver(() =&gt; {
      const isDark = document.documentElement.classList.contains(&#39;dark&#39;);
      localStorage.setItem(&#39;theme&#39;, isDark ? &#39;dark&#39; : &#39;light&#39;);
    });
    observer.observe(document.documentElement, {
      attributes: true,
      attributeFilter: [&#39;class&#39;]
    });
  }
&lt;/script&gt;
 
&lt;html lang=&quot;en&quot;&gt;
    &lt;body&gt;
      &lt;h1&gt;Astro&lt;/h1&gt;
    &lt;/body&gt;
&lt;/html&gt;
&lt;/script&gt;</code></pre>
